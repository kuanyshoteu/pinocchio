<script type="text/javascript">
    $('#bilimtap_send_query').click(function(e) {
        url = 'https://www.bilimtap.kz{{instance.connect_site_api}}'
        name = $('.bilimtap_query_name').val()
        phone = $('.bilimtap_query_phone').val()
        if (name.length > 0 && phone.length > 0) {
            $('.bilimtap_load').show()
            $('.bilimtap_success_load').hide()
            $(this).hide()
            $.ajax({
                url: url,
                data: {
                    'name':name,
                    'phone':phone,
                },
                dataType: 'json',
                success: function (data) {
                    $('.bilimtap_load').hide()
                    $('.bilimtap_success_load').show()
                }
            })          
        }
    })        
</script>
